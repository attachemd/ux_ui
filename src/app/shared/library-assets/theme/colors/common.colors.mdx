The common color palette provides a wide range of colors for various UI elements. Each color family contains shades from 50 (lightest) to 950 (darkest).

export const colorNames = [
  'red', 'orange', 'amber', 'yellow', 'lime',
  'green', 'emerald', 'teal', 'cyan', 'sky',
  'blue', 'indigo', 'violet', 'purple', 'fuchsia',
  'pink', 'rose', 'slate', 'gray', 'zinc', 'neutral', 'stone'
];

export const colorVariations = [
  '50', '100', '200', '300', '400',
  '500', '600', '700', '800', '900', '950'
];

export const PaletteCard = ({ color }) => (
  <div class="palette-card">
    <div class="palette-header">
      <h3 class="palette-title">{color}</h3>
    </div>
    <div class="palette-content">
      {colorVariations.map((variation) => (
        <div class="shade-row" key={variation} title={`--color-${color}-${variation}`}>
          <div class="shade-swatch" style={{ backgroundColor: `var(--color-${color}-${variation})` }}></div>
          <span class="shade-label">{variation}</span>
          <code class="shade-var">--color-{color}-{variation}</code>
          {(variation === '500' || variation === '600') && <span class="primary-badge">Base</span>}
        </div>
      ))}
    </div>
  </div>
);
