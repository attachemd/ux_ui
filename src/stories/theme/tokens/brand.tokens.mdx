  <div className="palette-card">
    <div className="palette-header"><h3 className="palette-title">{title}</h3></div>
    <div className="palette-content">
      {shades.map(s => (
        <div className="shade-row" key={s.key}>
          <div className="shade-swatch" style={{ backgroundColor: s.cssVar }}></div>
          <span className="shade-key">{s.key}</span>
          <code className="shade-varname">{s.varName}</code>

      ))}
    </div>
  </div>
);
